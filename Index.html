  <!-- Script: countdown + subscribe -->
  <script>
    // Launch date (Cairo time) — 1 Jan 2027 00:00:00 +02:00
    const launchDate = new Date('2027-01-01T00:00:00+02:00').getTime();

    function updateCountdown() {
      const now = Date.now();
      const diff = launchDate - now;

      if (diff <= 0) {
        document.getElementById('countdown').innerHTML = '<div style="font-weight:700; color:var(--gold);">We have launched! — انطلقنا!</div>';
        clearInterval(interval);
        return;
      }

      const days = Math.floor(diff / (1000 * 60 * 60 * 24));
      const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
      const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
      const seconds = Math.floor((diff % (1000 * 60)) / 1000);

      document.getElementById('days').textContent = days.toLocaleString();
      document.getElementById('hours').textContent = String(hours).padStart(2,'0');
      document.getElementById('minutes').textContent = String(minutes).padStart(2,'0');
      document.getElementById('seconds').textContent = String(seconds).padStart(2,'0');
    }

    const interval = setInterval(updateCountdown, 1000);
    updateCountdown();

    // Simple subscribe handler (no backend) — stores in localStorage as demo
    function handleSubscribe(e) {
      e.preventDefault();
      const email = document.getElementById('email').value.trim();
      if (!email) return;

      // store demo list locally (you can replace this with a server endpoint later)
      const listKey = 'kemet_subscribers_demo';
      const existing = JSON.parse(localStorage.getItem(listKey) || '[]');
      existing.push({email: email, at: new Date().toISOString()});
      localStorage.setItem(listKey, JSON.stringify(existing));

      const msg = document.getElementById('subMsg');
      msg.style.display = 'block';
      msg.textContent = 'شكراً! تم التسجيل. · Thank you — you are subscribed.';
      document.getElementById('subscribeForm').reset();

      // gentle animation
      msg.animate([{opacity:0, transform:'translateY(-6px)'},{opacity:1, transform:'translateY(0)'}], {duration:400, easing:'ease-out'});
      return false;
    }
  </script>
